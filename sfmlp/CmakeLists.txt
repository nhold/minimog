# minimog - sfmlp lib

file(GLOB includefiles "include/*.hpp")
file(GLOB sourcefiles "source/*.cpp")

add_library(sfmlp ${includefiles} ${sourcefiles})

# We require OpenGL and SFML.
find_package(OpenGL REQUIRED)
find_package(SFML COMPONENTS main window graphics network system audio)

#SFML linking
target_link_libraries(sfmlp sfml-graphics sfml-audio sfml-network ${OPENGL_LIBRARIES})

target_include_directories(sfmlp
    PUBLIC "include/"
    PRIVATE "${CMAKE_SOURCE_DIR}/thirdparty/json/"
    PRIVATE "${CMAKE_SOURCE_DIR}/thirdparty/meta/")

set_property(TARGET sfmlp PROPERTY CXX_STANDARD 20)
